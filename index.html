using UnityEngine; public class PlayerController : MonoBehaviour { public CharacterController controller; public float speed = 6f; public float sprintSpeed = 9f; public Transform ball; void Update() { MovePlayer(); HandleBallActions(); } void MovePlayer() { float x = Input.GetAxis("Horizontal"); float z = Input.GetAxis("Vertical"); Vector3 move = transform.right * x + transform.forward * z; float currentSpeed = Input.GetKey(KeyCode.LeftShift) ? sprintSpeed : speed; controller.Move(move * currentSpeed * Time.deltaTime); } void HandleBallActions() { if (Vector3.Distance(transform.position, ball.position) < 2f) { if (Input.GetKeyDown(KeyCode.Space)) { Shoot(); } if (Input.GetKeyDown(KeyCode.E)) { Pass(); } } } void Shoot() { ball.GetComponent().AddForce(transform.forward * 12f + Vector3.up * 2f, ForceMode.Impulse); } void Pass() { ball.GetComponent().AddForce(transform.forward * 7f, ForceMode.Impulse); } }
