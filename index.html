using UnityEngine; public class EliteFootballGameStarter : MonoBehaviour { [Header("Player Settings")] public CharacterController controller; public Transform ball; public float moveSpeed = 6f; public float sprintSpeed = 9f; public float turnSmoothTime = 0.08f; public float shootForce = 12f; public float passForce = 7f; [Header("Ground Check")] public Transform groundCheck; public float groundDistance = 0.4f; public LayerMask groundMask; private float turnSmoothVelocity; private Vector3 velocity; private bool isGrounded; private float gravity = -9.81f; private Rigidbody ballRb; void Start() { if (ball != null) { ballRb = ball.GetComponent(); } } void Update() { HandleMovement(); HandleBallActions(); } void HandleMovement() { isGrounded = Physics.CheckSphere(groundCheck.position, groundDistance, groundMask); if (isGrounded && velocity.y < 0) { velocity.y = -2f; } float horizontal = Input.GetAxisRaw("Horizontal"); float vertical = Input.GetAxisRaw("Vertical"); Vector3 direction = new Vector3(horizontal, 0f, vertical).normalized; if (direction.magnitude >= 0.1f) { float targetAngle = Mathf.Atan2(direction.x, direction.z) * Mathf.Rad2Deg; float angle = Mathf.SmoothDampAngle(transform.eulerAngles.y, targetAngle, ref turnSmoothVelocity, turnSmoothTime); transform.rotation = Quaternion.Euler(0f, angle, 0f); float speed = Input.GetKey(KeyCode.LeftShift) ? sprintSpeed : moveSpeed; controller.Move(direction * speed * Time.deltaTime); } velocity.y += gravity * Time.deltaTime; controller.Move(velocity * Time.deltaTime); } void HandleBallActions() { if (ballRb == null) return; float distanceToBall = Vector3.Distance(transform.position, ball.position); if (distanceToBall < 2.2f) { if (Input.GetKeyDown(KeyCode.Space)) { ShootBall(); } if (Input.GetKeyDown(KeyCode.E)) { PassBall(); } } } void ShootBall() { ballRb.linearVelocity = Vector3.zero; Vector3 shootDirection = transform.forward + Vector3.up * 0.15f; ballRb.AddForce(shootDirection.normalized * shootForce, ForceMode.Impulse); Debug.Log("Shot taken"); } void PassBall() { ballRb.linearVelocity = Vector3.zero; Vector3 passDirection = transform.forward; ballRb.AddForce(passDirection.normalized * passForce, ForceMode.Impulse); Debug.Log("Pass made"); } }
